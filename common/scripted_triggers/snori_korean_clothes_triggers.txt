snori_character_westernize_trigger = {
	exists = scope:character
	scope:character = {
		owner = {
			has_variable = snori_clothes_decision_taken
		}
	}
}

snori_pop_westernize_trigger = {
	exists = scope:pop
	scope:pop = {
		owner = {
			has_variable = snori_clothes_decision_taken
		}
	}
}

snori_korean_character_trigger = {
	exists = scope:character
	scope:character = {
		owner = {
			country_has_primary_culture = cu:korean
		}
	}
}
